~~~C++
#include<bits/stdc++.h>
using namespace std;
int main()
{
    string A,DA,B,DB;
    cin>>A>>DA>>B>>DB;
    int count1=0;
    int count2=0;
    string A1;
    string A2;
    int num1=0;
    int num2=0;
    for(int i=0;i<A.size();i++)
    {
        if(A[i]==DA[0])
            count1++;
    }
    for(int i=0;i<count1;i++)
    {
        A1[i]=DA[0];
    }
     for(int i=0;i<B.size();i++)
    {
        if (B[i] == DB[0])
            count2++;
    }
    for(int i=0;i<count2;i++)
    {
        A2[i] = DB[0];
    }
    if(count1)num1=stoi(A1);    //字符串转int  StringTOInt
    if(count2)num2=stoi(A2);
    cout<<num1+num2;
    return 0;
}
~~~
