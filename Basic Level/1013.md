~~~C++
#include<bits/stdc++.h>
using namespace std;

bool isPrime(int k)
{
    for(int i=2;i<=sqrt(k);i++)
    {
        if(k%i==0)
            return 0; 
   }
   return 1;
}


int main()
{
    int m,n;
    int num=2;
    cin>>m>>n;
    int cnt=0;
    vector<int> v;
    while(cnt<n)    //用边界条件限制while
    {
        if(isPrime(num))
        {
            cnt++;
            if(cnt>=m)
            {
                v.push_back(num);   //vector中加入数据
            }
        }
        num++;
    }
    for(int i=0;i<v.size();i++)
    {
        if(i%10!=0)     cout<<" ";
        cout<<v[i];
        if((i+1)%10==0)     cout<<endl;
    }

    return 0;
}
~~~
