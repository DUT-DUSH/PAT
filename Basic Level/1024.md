~~~C++
#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin >> s;
    int i = 0;
    while (s[i] != 'E')
        i++;
    int n = stoi(s.substr(i + 1));      //即i+1到末尾
    if (s[0] == '-')
        cout << "-";
    if (n < 0)
    {
        cout << "0.";
        for (int j = 0; j < abs(n) - 1; j++)
            cout << "0";
        for (int k = 1; k < i; k++)
        {
            if (k != 2)
                cout << s[k];
        }
    }
    else
    {
        for (int k = 1; k < i; k++)
        {
            if (k != 2)
                cout << s[k];
        }
        for (int k = 0; k < n - i + 3; k++)
            cout << "0";
    }
    return 0;
}
~~~
